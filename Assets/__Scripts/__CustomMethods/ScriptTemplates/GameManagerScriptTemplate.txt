using UnityEngine;

namespace SO
{
    [CreateAssetMenu(menuName = "SO/#SCRIPTNAME#", fileName = "#SCRIPTNAME#")]
    public class #SCRIPTNAME# : ScriptableObject
    {
        #region QOL向上処理
        // CakeParamsSOが保存してある場所のパス
        public const string PATH = "#SCRIPTNAME#";

        // CakeParamsSOの実体
        private static #SCRIPTNAME# _entity = null;
        public static #SCRIPTNAME# Entity
        {
            get
            {
                // 初アクセス時にロードする
                if (_entity == null)
                {
                    _entity = Resources.Load<#SCRIPTNAME#>(PATH);

                    //ロード出来なかった場合はエラーログを表示
                    if (_entity == null)
                    {
                        Debug.LogError(PATH + " not found");
                    }
                }

                return _entity;
            }
        }
        #endregion
    }
}
